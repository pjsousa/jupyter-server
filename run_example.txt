
# BUILD
#============================

docker build -t pedro/datascience-notebook .



# RUN
#============================

docker run -v /docker-volumes/data/ipython-server:/home/jovyan/work --name=ipython-server -d pedro/datascience-notebook

docker run -v /docker-volumes/data/ipython-server:/home/jovyan/work --name=ipython-server -d -e USE_HTTPS=yes -e PASSWORD="tr|4ngle#rs" pedro/datascience-notebook

docker run -v /docker-volumes/data/ipython-notebooks:/home/jovyan/work --name=ipython-server -d -e PASSWORD="tr|4ngle#rs" pedro/datascience-notebook


# HOST FILESYSTEM
#============================
cp ipython-nginx.conf /docker-volumes/config/nginx

mkdir /docker-volumes/config/nginx_ssl/ipython-server
mkdir /docker-volumes/config/nginx_ssl/ipython-server/certs
mkdir /docker-volumes/config/nginx_ssl/ipython-server/private

cp certs/*.crt /docker-volumes/config/nginx_ssl/ipython-server/certs
cp certs/*.key /docker-volumes/config/nginx_ssl/ipython-server/private
cp certs/*.pem /docker-volumes/config/nginx_ssl/ipython-server/private
